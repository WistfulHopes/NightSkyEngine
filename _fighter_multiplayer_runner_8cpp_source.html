28void AFighterMultiplayerRunner::BeginPlay()
29{
30 Super::BeginPlay();
31 GGPOSessionCallbacks cb = CreateCallbacks();
32 connectionManager = new RpcConnectionManager();

34 GGPONet::ggpo_set_disconnect_timeout(ggpo, 45000);
35 GGPONet::ggpo_set_disconnect_notify_start(ggpo, 15000);
36 for (int i = 0; i < 2; i++)
37 {
38 GGPOPlayerHandle handle;
39 GGPOPlayer* player = new GGPOPlayer();
40 player->type = GGPO_PLAYERTYPE_REMOTE;
41 if (i == GameState->GameInstance->PlayerIndex)
42 {
43 player->type = GGPO_PLAYERTYPE_LOCAL;
44 connectionManager->playerIndex = i == 0 ? 1 : 0;
45 }
46 player->player_num = i + 1;
47 player->connection_id = i;
48 GGPONet::ggpo_add_player(ggpo, player, &handle);
49 if (player->type == GGPO_PLAYERTYPE_LOCAL)
50 GGPONet::ggpo_set_frame_delay(ggpo, handle, 2);
51 // TODO: Logic for framedelay (based on ping? or make user choose?)
52 Players.Add(player);
53 PlayerInputIndex.Add(-1);
54 PlayerHandles.Add(handle);
55 }
56 GGPONet::ggpo_try_synchronize_local(ggpo);
57}

318void AFighterMultiplayerRunner::GgpoUpdate()
319{
320 GGPOErrorCode result = GGPOErrorCode::GGPO_OK;
321 int disconnect_flags;
322 int inputs[2] = {0};
323 for (int i = 0; i < 2; ++i)
324 {
325 if (Players[i]->type == GGPO_PLAYERTYPE_LOCAL)
326 {
327 int input = 0;
328 if (PlayerInputIndex[i] == -1)
329 {
330 for (int index = 0; index < 4; index++)
331 {
332 input |= GameState->GetLocalInputs(index);
333 }
334 }
335 else
336 {
337 input = GameState->GetLocalInputs(PlayerInputIndex[i]);
338 }
339
340 result = GGPONet::ggpo_add_local_input(ggpo, PlayerHandles[i], &input, sizeof(input));
341 }
342 }
343 if (GGPO_SUCCEEDED(result))
344 {
345 result = GGPONet::ggpo_synchronize_input(ggpo, (void*)inputs, sizeof(int) * 2, &disconnect_flags);
346 if (GGPO_SUCCEEDED(result))
347 {
348 GameState->UpdateGameState(inputs[0], inputs[1], false);
349 GGPONet::ggpo_advance_frame(ggpo);
350 }
351 }
352}

389AFighterMultiplayerRunner::fletcher32_checksum(short* data, size_t len)
390{
391 int sum1 = 0xffff, sum2 = 0xffff;
392
393 while (len)
394 {
395 size_t tlen = len > 360 ? 360 : len;
396 len -= tlen;
397 do
398 {
399 sum1 += *data++;
400 sum2 += sum1;
401 }
402 while (--tlen);
403 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
404 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
405 }
406
407 /* Second reduction step to reduce sums to 16 bits */
408 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
409 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
410 return sum2 << 16 | sum1;
411}
