39 if (i == GameState->GameInstance->PlayerIndex)

42 connectionManager->playerIndex = i == 0 ? 1 : 0;
43 }
44 player->player_num = i + 1;
45 player->connection_id = i;
46 GGPONet::ggpo_add_player(ggpo, player, &handle);
47 if (player->type == GGPO_PLAYERTYPE_LOCAL)
48 GGPONet::ggpo_set_frame_delay(ggpo, handle, 2);
49 // TODO: Logic for framedelay (based on ping? or make user choose?)
50 Players.Add(player);
51 PlayerInputIndex.Add(-1);
52 PlayerHandles.Add(handle);
53 }
54 GGPONet::ggpo_try_synchronize_local(ggpo);
55}

301void AFighterMultiplayerRunner::GgpoUpdate()
302{
303 GGPOErrorCode result = GGPOErrorCode::GGPO_OK;
304 int disconnect_flags;
305 int inputs[2] = {0};
306 for (int i = 0; i < 2; ++i)
307 {
308 if (Players[i]->type == GGPO_PLAYERTYPE_LOCAL)
309 {
310 int input = 0;
311 if (PlayerInputIndex[i] == -1)
312 {
313 for (int index = 0; index < 4; index++)
314 {
315 input |= GameState->GetLocalInputs(index);
316 }
317 }
318 else
319 {
320 input = GameState->GetLocalInputs(PlayerInputIndex[i]);
321 }
322
323 result = GGPONet::ggpo_add_local_input(ggpo, PlayerHandles[i], &input, sizeof(input));
324 }
325 }
326 if (GGPO_SUCCEEDED(result))
327 {
328 result = GGPONet::ggpo_synchronize_input(ggpo, (void*)inputs, sizeof(int) * 2, &disconnect_flags);
329 if (GGPO_SUCCEEDED(result))
330 {
331 GameState->UpdateGameState(inputs[0], inputs[1], false);
332 GGPONet::ggpo_advance_frame(ggpo);
333 }
334 }

372AFighterMultiplayerRunner::fletcher32_checksum(short* data, size_t len)
373{
374 int sum1 = 0xffff, sum2 = 0xffff;
375
376 while (len)
377 {
378 size_t tlen = len > 360 ? 360 : len;
379 len -= tlen;
380 do
381 {
382 sum1 += *data++;
383 sum2 += sum1;
384 }
385 while (--tlen);
386 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
387 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
388 }
389
390 /* Second reduction step to reduce sums to 16 bits */
391 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
392 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
393 return sum2 << 16 | sum1;
394}
