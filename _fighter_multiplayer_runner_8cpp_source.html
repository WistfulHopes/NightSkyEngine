45 player->connection_id = i;
46 GGPONet::ggpo_add_player(ggpo, player, &handle);
47 if(player->type == GGPO_PLAYERTYPE_LOCAL)
48 GGPONet::ggpo_set_frame_delay(ggpo,handle,2);// TODO: Logic for framedelay (based on ping? or make user choose?)
49 Players.Add(player);
50 PlayerInputIndex.Add(-1);
51 PlayerHandles.Add(handle);
52 }
53 GGPONet::ggpo_try_synchronize_local(ggpo);

305void AFighterMultiplayerRunner::GgpoUpdate()
306{
307 GGPOErrorCode result = GGPOErrorCode::GGPO_OK;
308 int disconnect_flags;
309 int inputs[2] = {0};
310 for (int i = 0; i < 2; ++i)
311 {
312 if (Players[i]->type == GGPO_PLAYERTYPE_LOCAL)
313 {
314 int input = 0;
315 if (PlayerInputIndex[i] == -1)
316 {
317 for (int index = 0; index < 4; index++)
318 {
319 input |= GameState->GetLocalInputs(index);
320 }
321 }
322 else
323 {
324 input = GameState->GetLocalInputs(PlayerInputIndex[i]);
325 }
326
327 result = GGPONet::ggpo_add_local_input(ggpo, PlayerHandles[i], &input, sizeof(input));
328 }
329 }
330 if (GGPO_SUCCEEDED(result))
331 {
332 result = GGPONet::ggpo_synchronize_input(ggpo, (void*)inputs, sizeof(int) * 2, &disconnect_flags);
333 if (GGPO_SUCCEEDED(result))
334 {
335 GameState->UpdateGameState(inputs[0], inputs[1], false);
336 GGPONet::ggpo_advance_frame(ggpo);
337 }
338 }
339}

376AFighterMultiplayerRunner::fletcher32_checksum(short* data, size_t len)
377{
378 int sum1 = 0xffff, sum2 = 0xffff;
379
380 while (len)
381 {
382 size_t tlen = len > 360 ? 360 : len;
383 len -= tlen;
384 do
385 {
386 sum1 += *data++;
387 sum2 += sum1;
388 }
389 while (--tlen);
390 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
391 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
392 }
393
394 /* Second reduction step to reduce sums to 16 bits */
395 sum1 = (sum1 & 0xffff) + (sum1 >> 16);
396 sum2 = (sum2 & 0xffff) + (sum2 >> 16);
397 return sum2 << 16 | sum1;
398}
