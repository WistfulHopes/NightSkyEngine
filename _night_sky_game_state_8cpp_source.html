292void ANightSkyGameState::MatchInit()
293{
294 if (FighterRunner)
295 {
296 FighterRunner->Destroy();
297 FighterRunner = nullptr;
298 }
299
300 FActorSpawnParameters SpawnParameters;
301 SpawnParameters.Owner = GetOwner();
302
303 switch (GameInstance->FighterRunner)
304 {
305 case LocalPlay:
306 FighterRunner = GetWorld()->SpawnActor<AFighterLocalRunner>(SpawnParameters);
307 break;
308 case Multiplayer:
309 if (GameInstance->IsReplay)
310 FighterRunner = GetWorld()->SpawnActor<AFighterReplayRunner>(SpawnParameters);
311 else
312 FighterRunner = GetWorld()->SpawnActor<AFighterMultiplayerRunner>(SpawnParameters);
313 break;
314 case SyncTest:
315 FighterRunner = GetWorld()->SpawnActor<AFighterSynctestRunner>(SpawnParameters);
316 break;

319 break;
320 }

323
324 BattleState.RandomManager = GameInstance->BattleData.Random;
325 BattleState.TeamData[0].TeamCount = GameInstance->BattleData.PlayerListP1.Num();
326 BattleState.TeamData[1].TeamCount = GameInstance->BattleData.PlayerListP2.Num();
327 BattleState.GaugeP1.Empty();
328 BattleState.GaugeP2.Empty();
329 BattleState.GaugeP1.AddDefaulted(BattleState.MaxGauge.Num());
330 BattleState.GaugeP2.AddDefaulted(BattleState.MaxGauge.Num());
331
332 CallBattleExtension(BattleExtension_MatchInit);
333
334 for (int i = 0; i < Players.Num(); i++)
335 {
336 Players[i]->PlayerIndex = i >= BattleState.TeamData[0].TeamCount;
337 Players[i]->TeamIndex = i >= BattleState.TeamData[0].TeamCount ? i - BattleState.TeamData[0].TeamCount : i;
338 Players[i]->PlayerFlags &= ~PLF_IsOnScreen;
339 Players[i]->ObjNumber = i + MaxBattleObjects;
340 Players[i]->CallSubroutine(Subroutine_Cmn_MatchInit);
341 Players[i]->CallSubroutine(Subroutine_MatchInit);
342 }
343 for (int i = 0; i < MaxBattleObjects; i++)
344 {
345 Objects[i]->ObjNumber = i;
346 }
347 for (int i = SortedObjects.Num() - 1; i >= 0; i--)
348 {
349 SetDrawPriorityFront(SortedObjects[i]);
350 }
351
352 BattleState.MainPlayer[0] = Players[0];
353 BattleState.MainPlayer[0]->PlayerFlags |= PLF_IsOnScreen;
354 BattleState.MainPlayer[1] = Players[BattleState.TeamData[0].TeamCount];
355 BattleState.MainPlayer[1]->PlayerFlags |= PLF_IsOnScreen;
356 BattleState.BattleFormat = GameInstance->BattleData.BattleFormat;
357 BattleState.MaxTimeUntilRoundStart = GameInstance->BattleData.TimeUntilRoundStart;
358 BattleState.MaxRoundCount = GameInstance->BattleData.RoundCount;
359 BattleState.RoundTimer = GameInstance->BattleData.StartRoundTimer * 60;
360
361 PlayMusic(GameInstance->BattleData.MusicName);
362 BattleHudActor->BottomWidget->PlayFadeOutAnim();
363 RoundInit();
364 PlayIntros();

367void ANightSkyGameState::UpdateGameState(int32 Input1, int32 Input2, bool bShouldResimulate)
368{
369 if (bShouldResimulate == false && bIsResimulating == true) RollbackStartAudio(BattleState.FrameNumber);
370 bIsResimulating = bShouldResimulate;
371 LocalFrame++;
372
373 ParticleManager->UpdateParticles();
374 UpdateScreen();
375
376 if (BattleState.BattlePhase == EBattlePhase::Intro)
377 {
378 if (BattleState.CurrentIntroSide == INT_P1)
379 {
380 if (GetMainPlayer(true)->IntroEndFlag)
381 {
382 GetMainPlayer(true)->JumpToStatePrimary(State_Universal_Stand);
383 }
384 if (GetMainPlayer(true)->GetCurrentStateName(StateMachine_Primary) != GetMainPlayer(true)->IntroName)
385 {
386 GetMainPlayer(false)->JumpToStatePrimary(GetMainPlayer(false)->IntroName);
387 BattleState.CurrentIntroSide = INT_P2;
388 }
389 }
390 else if (BattleState.CurrentIntroSide == INT_P2)
391 {
392 if (GetMainPlayer(false)->IntroEndFlag)
393 {
394 GetMainPlayer(false)->JumpToStatePrimary(State_Universal_Stand);
395 }
396 if (GetMainPlayer(false)->GetCurrentStateName(StateMachine_Primary) != GetMainPlayer(false)->IntroName)
397 {
398 GetMainPlayer(true)->JumpToStatePrimary(State_Universal_Stand);
399 GetMainPlayer(false)->JumpToStatePrimary(State_Universal_Stand);
400 BattleState.CurrentIntroSide = INT_None;
401 BattleState.BattlePhase = EBattlePhase::Battle;
402 BattleHudVisibility(true);
403 BattleHudActor->BottomWidget->PlayRoundInitAnim();
404 }
405 }
406 }
407
408 if (!GameInstance->IsTraining && !BattleState.PauseTimer)
409 {
410 if (BattleState.TimeUntilRoundStart > 0)
411 BattleState.TimeUntilRoundStart--;
412 else
413 BattleState.RoundTimer--;
414 }
415
416 if (BattleState.RoundTimer < 0)
417 BattleState.RoundTimer = 0;
418 BattleState.FrameNumber++;
419
420 if (!BattleState.SuperFreezeDuration)
421 {
422 CallBattleExtension(BattleExtension_Update);
423 for (int i = 0; i < 2; i++)
424 {
425 if (GetMainPlayer(i == 0)->ComboCounter) continue;
426 for (auto& Cooldown : BattleState.TeamData[i].CooldownTimer)
427 {
428 if (Cooldown == 0) continue;
429 if (GameInstance->IsTraining) Cooldown = 0;
430 else Cooldown--;
431 }
432 }
433 }
434
435 if (BattleState.CurrentSequenceTime != -1)
436 BattleState.CurrentSequenceTime++;
437
438 SortObjects();
439
440 if (!BattleState.MainPlayer[0]->bIsCpu) BattleState.MainPlayer[0]->Inputs = Input1;
441 else if (const auto CPU = Cast<ANightSkyAIController>(BattleState.MainPlayer[0]->Controller); CPU)
442 CPU->Update();
443 if (!BattleState.MainPlayer[1]->bIsCpu) BattleState.MainPlayer[1]->Inputs = Input2;
444 else if (const auto CPU = Cast<ANightSkyAIController>(BattleState.MainPlayer[1]->Controller); CPU)
445 CPU->Update();
446 for (int i = 0; i < Players.Num(); i++)
447 {
448 for (int j = 0; j < Players.Num(); j++)
449 {
450 if (i < BattleState.TeamData[0].TeamCount)
451 {
452 Players[i]->Enemy = GetMainPlayer(false);
453 }
454 else
455 {
456 Players[i]->Enemy = GetMainPlayer(true);
457 }
458 }
459 }
460
461 HandleHitCollision();
462
463 for (int i = 0; i < SortedObjects.Num(); i++)
464 {
465 if (i == BattleState.ActiveObjectCount)
466 break;
467 if (((BattleState.SuperFreezeSelfDuration && SortedObjects[i] == BattleState.SuperFreezeCaller)
468 || (BattleState.SuperFreezeDuration && SortedObjects[i] != BattleState.SuperFreezeCaller))
469 && (SortedObjects[i]->MiscFlags & MISC_IgnoreSuperFreeze) == 0)
470 {
471 if (SortedObjects[i]->IsPlayer)
472 {
473 if (SortedObjects[i]->Player->PlayerFlags & PLF_IsStunned)
474 SortedObjects[i]->Player->HandleBufferedState();
475 SortedObjects[i]->GetBoxes();
476 SortedObjects[i]->Player->StoredInputBuffer.Update(SortedObjects[i]->Player->Inputs);
477
478 // Handle state transitions
479 SortedObjects[i]->Player->HandleStateMachine(true, SortedObjects[i]->Player->GetStateMachine(StateMachine_Primary));
480 for (auto& StateMachine : SortedObjects[i]->Player->SubStateMachines)
481 {
482 SortedObjects[i]->Player->HandleStateMachine(true, StateMachine);
483 }
484 }
485 continue;
486 }
487 SortedObjects[i]->Update();
488 }
489
490 if (BattleState.SuperFreezeSelfDuration == 1)
491 {
492 BattleState.SuperFreezeCaller->TriggerEvent(EVT_SuperFreezeEnd, StateMachine_Primary);
493 }
494 if (BattleState.SuperFreezeDuration == 1)
495 {
496 for (int i = 0; i < SortedObjects.Num(); i++)
497 {
498 SortedObjects[i]->TriggerEvent(EVT_SuperFreezeEnd, StateMachine_Primary);
499 }
500
501 BattleState.PauseTimer = false;
502 BattleState.SuperFreezeCaller = nullptr;
503 BattleHudVisibility(true);
504 }
505
506 if (BattleState.SuperFreezeSelfDuration) BattleState.SuperFreezeSelfDuration--;
507 if (BattleState.SuperFreezeDuration) BattleState.SuperFreezeDuration--;
508
509 HandlePushCollision();
510 SetScreenBounds();
511 HandleRoundWin();
512
513 // these aren't strictly game state related, but tying them to game state update makes things better
514
515 if (GameInstance->FighterRunner == Multiplayer && !GameInstance->IsReplay)
516 {
517 GameInstance->UpdateReplay(Input1, Input2);
518 }
519
520 CollisionView();
521
522 const auto [network, timesync] = GetNetworkStats();
523 NetworkStats.Ping = network.ping;
524 const int32 LocalFramesBehind = timesync.local_frames_behind;
525 const int32 RemoteFramesBehind = timesync.remote_frames_behind;
526
527 if (LocalFramesBehind < 0 && RemoteFramesBehind < 0)
528 {
529 NetworkStats.RollbackFrames = abs(abs(LocalFramesBehind) - abs(RemoteFramesBehind));
530 }
531 else if (LocalFramesBehind > 0 && RemoteFramesBehind > 0)
532 {
533 NetworkStats.RollbackFrames = 0;
534 }
535 else
536 {
537 NetworkStats.RollbackFrames = abs(LocalFramesBehind) + abs(RemoteFramesBehind);
538 }
539
540 for (int i = 0; i < 2; i++)
541 {
542 if (BattleState.Meter[i] > BattleState.MaxMeter[i])
543 BattleState.Meter[i] = BattleState.MaxMeter[i];
544 if (BattleState.Meter[i] < 0)
545 BattleState.Meter[i] = 0;
546 }
547
548 for (int i = 0; i < BattleState.MaxGauge.Num(); i++)
549 {
550 if (BattleState.GaugeP1[i] > BattleState.MaxGauge[i])
551 BattleState.GaugeP1[i] = BattleState.MaxGauge[i];
552 if (BattleState.GaugeP1[i] < 0)
553 BattleState.GaugeP1[i] = 0;
554 }
555 for (int i = 0; i < BattleState.MaxGauge.Num(); i++)
556 {
557 if (BattleState.GaugeP2[i] > BattleState.MaxGauge[i])
558 BattleState.GaugeP2[i] = BattleState.MaxGauge[i];
559 if (BattleState.GaugeP2[i] < 0)
560 BattleState.GaugeP2[i] = 0;
561 }
562
563 ParticleManager->PauseParticles();
564 UpdateVisuals();
565 UpdateCamera();
566 UpdateHUD();
567 ManageAudio();
568}

611 if (Target->B < ScreenData->ObjBottom * 1000) ScreenData->ObjBottom = Target->B / 1000;
612 if (Target->B > ScreenData->HigherObjBottom * 1000) ScreenData->HigherObjBottom = Target->B / 1000;
613 if (TargetL < ScreenData->ObjLeft * 1000) ScreenData->ObjLeft = TargetL / 1000;

617

619 ScreenData->ObjHeight = ScreenData->ObjTop - ScreenData->ObjBottom;

622void ANightSkyGameState::UpdateScreen()
623{
624 const auto ScreenData = &BattleState.ScreenData;
625
626 ScreenData->MaxZoomOutWidth = ScreenData->DefaultMaxWidth;
627 ScreenData->ZoomOutBeginX = ScreenData->DefaultWidth;
628
629 SetScreenCorners();
630
631 if ((ScreenData->Flags & SCR_Lock) == 0)
632 {
633 if ((ScreenData->Flags & SCR_LockXPos) == 0)
634 ScreenData->TargetCenterX = ScreenData->ObjLeft + ScreenData->
635 ObjLength / 2;
636
637 if ((ScreenData->Flags & SCR_LockWidth) == 0)
638 {
639 ScreenData->TargetWidth = ScreenData->ObjLength + ScreenData->ZoomOutBeginX / 4;
640 if (ScreenData->TargetWidth < ScreenData->ZoomOutBeginX)
641 ScreenData->TargetWidth = ScreenData->
642 ZoomOutBeginX;
643 if (ScreenData->TargetWidth > ScreenData->MaxZoomOutWidth)
644 ScreenData->TargetWidth = ScreenData->
645 MaxZoomOutWidth;
646 }
647
648 if ((ScreenData->Flags & SCR_LockYPos) == 0)
649 {
650 int TargetOffsetY;
651
652 auto Ratio = ScreenData->DefaultWidth * 1000 / ScreenData->TargetWidth;
653
654 if (Ratio * ScreenData->ObjBottom / 1000 <= ScreenData->TargetOffsetAirYPos ||
655 Ratio * (ScreenData->HigherObjBottom - ScreenData->ObjBottom) / 1000 >= ScreenData->
656 TargetOffsetAirYDist)
657 {
658 TargetOffsetY = FMath::Min(ScreenData->TargetOffsetY + ScreenData->TargetOffsetLandYAdd, ScreenData->TargetOffsetLandYMax);
659 }
660 else
661 {
662 TargetOffsetY = FMath::Max(ScreenData->TargetOffsetY - ScreenData->TargetOffsetAirYAdd, ScreenData->TargetOffsetAirYMax);
663 }
664
665 ScreenData->TargetOffsetY = TargetOffsetY;
666
667 Ratio = Ratio * ScreenData->HigherObjBottom / 1000 - Ratio * TargetOffsetY / 1000;
668 ScreenData->TargetCenterY = FMath::Clamp(Ratio, 0, ScreenData->StageBoundsTop);
669 }
670 }
671
672 const auto Width = ScreenData->TargetWidth - ScreenData->ScreenWorldWidth;
673 ScreenData->WidthVelocity = Width / 14 - 1;
674 ScreenData->ScreenWorldWidth += ScreenData->WidthVelocity;
675
676 auto ScreenXSpeedFrame = 10;
677 auto ScreenXSpeed = 50;
678

970{

972 if (BattleState.BattlePhase == EBattlePhase::EndScreen) return true;
973 if (BattleState.CurrentWinSide == WIN_P2)
974 {

977 EndMatch();
978 }
979 }
980 else
981 {
982 if (GetMainPlayer(true)->RoundEndFlag == true)
983 {
984 EndMatch();
985 }
986 }
987 return true;
988 }
989 switch (BattleState.BattleFormat)
990 {
991 case EBattleFormat::Rounds:
992 {
993 if (BattleState.P1RoundsWon >= BattleState.MaxRoundCount && BattleState.P2RoundsWon < BattleState.P1RoundsWon)
994 {
995 GetMainPlayer(true)->JumpToStatePrimary(State_Universal_MatchWin);

1001 {
1002 GetMainPlayer(false)->JumpToStatePrimary(State_Universal_MatchWin);
1003 GameInstance->EndRecordReplay();
1004 BattleState.CurrentWinSide = WIN_P2;
1005 return true;
1006 }
1007 if (BattleState.P1RoundsWon > BattleState.MaxRoundCount && BattleState.P2RoundsWon > BattleState.MaxRoundCount)
1008 {
1009 GameInstance->EndRecordReplay();
1010 BattleState.CurrentWinSide = WIN_Draw;
1011 return true;
1012 }
1013 }
1014 return false;
1015 case EBattleFormat::Tag:
1016 case EBattleFormat::KOF:
1017 {
1018 if (BattleState.P1RoundsWon >= BattleState.TeamData[1].TeamCount && BattleState.P2RoundsWon < BattleState.P1RoundsWon)
1019 {
1020 GetMainPlayer(true)->JumpToStatePrimary(State_Universal_MatchWin);
1021 GameInstance->EndRecordReplay();
1022 BattleState.CurrentWinSide = WIN_P1;
1023 return true;
1024 }
1025 if (BattleState.P2RoundsWon >= BattleState.TeamData[0].TeamCount && BattleState.P1RoundsWon < BattleState.P2RoundsWon)
1026 {
1027 GetMainPlayer(false)->JumpToStatePrimary(State_Universal_MatchWin);
1028 GameInstance->EndRecordReplay();
1029 BattleState.CurrentWinSide = WIN_P2;
1030 return true;
1031 }
1032 if (BattleState.P1RoundsWon >= BattleState.TeamData[1].TeamCount && BattleState.P2RoundsWon >= BattleState.TeamData[0].TeamCount)
1033 {
1034 GameInstance->EndRecordReplay();
1035 BattleState.CurrentWinSide = WIN_Draw;
1036 return true;
1037 }
1038 }
1039 return false;
1040 default:
1041 return false;
1042 }
