245void ANightSkyGameState::UpdateGameState(int32 Input1, int32 Input2)
246{
247 LocalFrame++;
248 ParticleManager->UpdateParticles();
249 if (!GameInstance->IsTraining && !BattleState.PauseTimer)
250 {
251 if (BattleState.TimeUntilRoundStart > 0)
252 BattleState.TimeUntilRoundStart--;
253 else
254 BattleState.RoundTimer--;
255 }
256
257 if (BattleState.RoundTimer < 0)
258 BattleState.RoundTimer = 0;
259 BattleState.FrameNumber++;
260
261 if (BattleState.CurrentSequenceTime != -1)
262 BattleState.CurrentSequenceTime++;
263
264 SortObjects();
265
266 Players[0]->Inputs = Input1;
267 Players[3]->Inputs = Input2;
268 for (int i = 0; i < MaxPlayerObjects; i++)
269 {
270 if (Players[i]->PlayerFlags & PLF_IsOnScreen)
271 {
272 for (int j = 0; j < MaxPlayerObjects; j++)
273 {
274 if (i < MaxPlayerObjects / 2)
275 {
276 if (j >= MaxPlayerObjects / 2 && Players[j]->PlayerFlags & PLF_IsOnScreen)
277 {
278 Players[i]->Enemy = Players[j];
279 }
280 }
281 else
282 {
283 if (j < MaxPlayerObjects / 2 && Players[j]->PlayerFlags & PLF_IsOnScreen)
284 {
285 Players[i]->Enemy = Players[j];
286 }
287 }
288 }
289 }
290 }
291
292 HandleHitCollision();
293
294 for (int i = 0; i < MaxBattleObjects + MaxPlayerObjects; i++)
295 {
296 if (i == BattleState.ActiveObjectCount)
297 break;
298 if ((BattleState.SuperFreezeSelfDuration && SortedObjects[i] == BattleState.SuperFreezeCaller)
299 || (BattleState.SuperFreezeDuration && SortedObjects[i] != BattleState.SuperFreezeCaller))
300 {
301 if (SortedObjects[i]->IsPlayer)
302 {
303 if (SortedObjects[i]->Player->PlayerFlags & PLF_IsStunned)
304 SortedObjects[i]->Player->HandleBufferedState();
305 SortedObjects[i]->Player->StoredInputBuffer.Tick(SortedObjects[i]->Player->Inputs);
306 SortedObjects[i]->Player->HandleStateMachine(true); //handle state transitions
307
308 }
309 SortedObjects[i]->UpdateVisuals();
310 continue;
311 }
312 if (!SortedObjects[i]->IsPlayer || SortedObjects[i]->Player->PlayerFlags & PLF_IsOnScreen)
313 SortedObjects[i]->Update();
314 }
315
316 if (BattleState.SuperFreezeSelfDuration == 1)
317 {
318 BattleState.SuperFreezeCaller->TriggerEvent(EVT_SuperFreezeEnd);
319 }
320 if (BattleState.SuperFreezeDuration == 1)
321 {
322 for (int i = 0; i < MaxBattleObjects + MaxPlayerObjects; i++)
323 {
324 SortedObjects[i]->TriggerEvent(EVT_SuperFreezeEnd);
325 }
326
327 BattleState.PauseTimer = false;
328 BattleHudVisibility(true);
329 }
330
331 if (BattleState.SuperFreezeSelfDuration) BattleState.SuperFreezeSelfDuration--;
332 if (BattleState.SuperFreezeDuration) BattleState.SuperFreezeDuration--;
333
334 HandlePushCollision();
335 SetScreenBounds();
336 SetStageBounds();
337 HandleRoundWin();
338 ParticleManager->PauseParticles();
339 if (GameInstance->FighterRunner == Multiplayer && !GameInstance->IsReplay)
340 {
341 GameInstance->UpdateReplay(Input1, Input2);
342 }
343 CollisionView();
344
345 const FGGPONetworkStats Network = GetNetworkStats();
346 NetworkStats.Ping = Network.network.ping;
347 const int32 LocalFramesBehind = Network.timesync.local_frames_behind;
348 const int32 RemoteFramesBehind = Network.timesync.remote_frames_behind;
349
350 if (LocalFramesBehind < 0 && RemoteFramesBehind < 0)
351 NetworkStats.RollbackFrames = abs(abs(LocalFramesBehind) - abs(RemoteFramesBehind));
352 else if (LocalFramesBehind > 0 && RemoteFramesBehind > 0)
353 NetworkStats.RollbackFrames = 0;
354 else
355 NetworkStats.RollbackFrames = abs(LocalFramesBehind) + abs(RemoteFramesBehind);
356
357 // these aren't strictly game state related, but tying them to game state update makes things better
358 UpdateCamera();
359 UpdateHUD();
360 ManageAudio();
361
362 CallBattleExtension("Update");
363
364 for (int i = 0; i < 2; i++)
365 {
366 if (BattleState.Meter[i] > BattleState.MaxMeter[i])
367 BattleState.Meter[i] = BattleState.MaxMeter[i];
368 if (BattleState.Meter[i] < 0)
369 BattleState.Meter[i] = 0;
370
371 for (int j = 0; j < GaugeCount; j++)
372 {
373 if (BattleState.Gauge[i][j] > BattleState.MaxGauge[j])
374 BattleState.Gauge[i][j] = BattleState.MaxGauge[j];
375 if (BattleState.Gauge[i][j] < 0)
376 BattleState.Gauge[i][j] = 0;
377 }
378 }

388void ANightSkyGameState::SortObjects()
389{
390 BattleState.ActiveObjectCount = MaxPlayerObjects;
391 for (int i = MaxPlayerObjects; i < MaxBattleObjects + MaxPlayerObjects; i++)
392 {
393 for (int j = i + 1; j < MaxBattleObjects + MaxPlayerObjects; j++)
394 {
395 if (SortedObjects[j]->IsActive && !SortedObjects[i]->IsActive)
396 {
397 ABattleObject* Temp = SortedObjects[i];
398 SortedObjects[i] = SortedObjects[j];
399 SortedObjects[j] = Temp;
400 }
401 }
402 if (SortedObjects[i]->IsActive)
403 {
404 BattleState.ActiveObjectCount++;
405 }
