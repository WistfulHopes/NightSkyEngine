40void ANightSkyWTGameState::Init(APlayerObject* P1, APlayerObject* P2)
41{
42 bIsBattling = true;
43
44 if (IsValid(BattleExtensionData))
45 {
46 for (auto Extension : BattleExtensionData->ExtensionArray)
47 {
48 BattleExtensions.Add(NewObject<UBattleExtension>(this, Extension));
49 BattleExtensions.Last()->Parent = this;
50 BattleExtensionNames.Add(BattleExtensions.Last()->Name);
51 }
52 }
53
54 {
55 Players.Add(P1);
56 P1->SetActorTransform(BattleSceneTransform);
57 P1->PlayerFlags |= PLF_IsOnScreen;
58 SortedObjects.Add(P1);
59 P1->InitPlayer();
60 P1->GameState = this;
61 }
62 {
63 Players.Add(P2);
64 P2->SetActorTransform(BattleSceneTransform);
65 P2->PlayerFlags |= PLF_IsOnScreen;
66 SortedObjects.Add(P2);
67 P2->InitPlayer();
68 P2->GameState = this;
69 }
70
71 for (int i = 0; i < MaxBattleObjects; i++)
72 {
73 Objects.Add(GetWorld()->SpawnActor<ABattleObject>(ABattleObject::StaticClass(), BattleSceneTransform));
74 Objects[i]->GameState = this;
75 SortedObjects.Add(Objects.Last());
76 }
77
78 const FVector NewCameraLocation = BattleSceneTransform.GetRotation().RotateVector(FVector(0, 1080, 175)) + BattleSceneTransform.GetLocation();
79 FRotator CameraRotation = BattleSceneTransform.GetRotation().Rotator();
80 CameraRotation.Yaw -= 90;
81
82 CameraActor->SetActorLocation(NewCameraLocation);
83 CameraActor->SetActorRotation(CameraRotation);
84 SequenceCameraActor->SetActorLocation(NewCameraLocation);
85
86 MatchInit();
87 HUDInit();
88}
