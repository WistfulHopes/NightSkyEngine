38void ABattleObject::Move()
39{
40 if (IsPlayer)
41 {

45 }

53 }
54
55 // Set previous pos values
56 PrevPosX = PosX;
57 PrevPosY = PosY;

59
60 if (BlendOffset && BlendCelName != FGameplayTag::EmptyTag && MaxCelTime)
61 {
62 const int32 TmpOffsetX = (NextOffsetX - PrevOffsetX) * (MaxCelTime - TimeUntilNextCel) / MaxCelTime;
63 const int32 TmpOffsetY = (NextOffsetY - PrevOffsetY) * (MaxCelTime - TimeUntilNextCel) / MaxCelTime;
64
65 AddPosXWithDir(TmpOffsetX);
66 PosY += TmpOffsetY;
67 }
68
69 SpeedX = SpeedX * SpeedXRatePerFrame / 100;
70 SpeedY = SpeedY * SpeedYRatePerFrame / 100;
71 SpeedZ = SpeedZ * SpeedZRatePerFrame / 100;
72 SpeedX = SpeedX * SpeedXRate / 100;
73 SpeedY = SpeedY * SpeedYRate / 100;
74 SpeedZ = SpeedZ * SpeedZRate / 100;
75
76 SpeedXRate = SpeedYRate = SpeedZRate = 100;
77
78 if (MiscFlags & MISC_InertiaEnable) //only use inertia if enabled
79 {
80 if (PosY <= GroundHeight && MiscFlags & MISC_FloorCollisionActive) //only decrease inertia if grounded
81 {
82 Inertia = Inertia - Inertia / 10;
83 }
84 if (Inertia > -875 && Inertia < 875) //if inertia small enough, set to zero
85 {
86 Inertia = 0;
87 }
88 AddPosXWithDir(Inertia);
89 }
90
91 if (IsPlayer)
92 {
93 int32 ModifiedPushback;
94 if (PosY > GroundHeight)
95 ModifiedPushback = Player->Pushback * 84;
96 else if (Player->Stance == ACT_Crouching)
97 ModifiedPushback = Player->Pushback * 86;
98 else
99 ModifiedPushback = Player->Pushback * 88;
100
101 Player->Pushback = ModifiedPushback / 100;
102
103 if (PosY <= GroundHeight || !(Player->PlayerFlags & PLF_IsStunned))
104 AddPosXWithDir(Player->Pushback);
105 }
106
107 AddPosXWithDir(SpeedX); //apply speed
108
109 if (IsPlayer && Player != nullptr)
110 {
111 if (Player->AirDashTimer == 0 || (SpeedY > 0 && ActionTime < 5))
112 // only set y speed if not airdashing/airdash startup not done
113 {
114 PosY += SpeedY;
115 if (PosY > GroundHeight || !(MiscFlags & MISC_FloorCollisionActive))
116 SpeedY -= Gravity;
117 }
118 else
119 {
120 SpeedY = 0;
121 }
122 }
123 else
124 {
125 PosY += SpeedY;
126 if (PosY > GroundHeight || !(MiscFlags & MISC_FloorCollisionActive))
127 SpeedY -= Gravity;
128 }
129
130 if (PosY < GroundHeight && MiscFlags & MISC_FloorCollisionActive) //if on ground, force y values to zero
131 {
132 PosY = GroundHeight;
133 }
134
135 PosZ += SpeedZ;
136}

344bool ABattleObject::SuperArmorSuccess(const ABattleObject* Attacker) const
345{
346 if (SuperArmorData.Type == ARM_None) return false;
347 if (SuperArmorData.ArmorHits == 0) return false;
348
349 if (SuperArmorData.bArmorMid && Attacker->HitCommon.BlockType == BLK_Mid) return true;
350 if (SuperArmorData.bArmorOverhead && Attacker->HitCommon.BlockType == BLK_High) return true;
351 if (SuperArmorData.bArmorLow && Attacker->HitCommon.BlockType == BLK_Low) return true;
352 if (SuperArmorData.bArmorStrike && Attacker->AttackFlags & ATK_HitActive && !(Attacker->AttackFlags &
353 ATK_AttackProjectileAttribute))
354 return true;
355 if (SuperArmorData.bArmorThrow && Attacker->IsPlayer && Attacker->Player->PlayerFlags & PLF_ThrowActive) return
356 true;
357 if (SuperArmorData.bArmorHead && Attacker->AttackFlags & ATK_HitActive && Attacker->AttackFlags &
358 ATK_AttackHeadAttribute)
359 return true;
360 if (SuperArmorData.bArmorProjectile && Attacker->AttackFlags & ATK_HitActive && Attacker->AttackFlags &
361 ATK_AttackProjectileAttribute)
362 return true;
363
364 return false;
365}

458 PosX += CollisionDepth / 2;

1477static void clip_line_y(
1478 const FVector2D& line_a, const FVector2D& line_b,
1479 float min_x, float max_x,
1480 float* min_y, float* max_y)
1481{
1482 const auto delta = line_b - line_a;
1483
1484 if (abs(delta.X) > FLT_EPSILON)
1485 {
1486 const auto slope = delta.Y / delta.X;
1487 const auto intercept = line_a.Y - slope * line_a.X;
1488 *min_y = slope * min_x + intercept;
1489 *max_y = slope * max_x + intercept;
1490 }
1491 else
1492 {
1493 *min_y = line_a.Y;
1494 *max_y = line_b.Y;
1495 }
1496
1497 if (*min_y > *max_y)
1498 std::swap(*min_y, *max_y);
1499}

1501bool line_box_intersection(
1502 const FVector2D& box_min, const FVector2D& box_max,
1503 const FVector2D& line_a, const FVector2D& line_b,
1504 float* entry_fraction, float* exit_fraction)
1505{
1506 // No intersection if line runs along the edge of the box
1507 if (line_a.X == line_b.X && (line_a.X == box_min.X || line_a.X == box_max.X))
1508 return false;
1509
1510 if (line_a.Y == line_b.Y && (line_a.Y == box_min.Y || line_a.Y == box_max.Y))
1511 return false;
1512
1513 // Clip X values to segment within box_min.X and box_max.X
1514 const auto min_x = max(min(line_a.X, line_b.X), box_min.X);
1515 const auto max_x = min(max(line_a.X, line_b.X), box_max.X);
1516
1517 // Check if the line is in the bounds of the box on the X axis
1518 if (min_x > max_x)
1519 return false;
1520
1521 // Clip Y values to segment within min_x and max_x
1522 float min_y, max_y;
1523 clip_line_y(line_a, line_b, min_x, max_x, &min_y, &max_y);
1524
1525 // Clip Y values to segment within box_min.Y and box_max.Y
1526 min_y = max(min_y, (float)box_min.Y);
1527 max_y = min(max_y, (float)box_max.Y);
1528
1529 // Check if the clipped line is in the bounds of the box on the Y axis
1530 if (min_y > max_y)
1531 return false;
1532
1533 const FVector2D entry(
1534 line_a.X < line_b.X ? min_x : max_x,
1535 line_a.Y < line_b.Y ? min_y : max_y);
1536
1537 const FVector2D exit(
1538 line_a.X > line_b.X ? min_x : max_x,
1539 line_a.Y > line_b.Y ? min_y : max_y);
1540
1541 const auto length = (line_b - line_a).Size();
1542 *entry_fraction = (entry - line_a).Size() / length;
1543 *exit_fraction = (exit - line_a).Size() / length;
1544
1545 return true;
1546}
