44void ANightSkyPlayerController::Tick(float DeltaTime)
45{
46 Super::Tick(DeltaTime);
47 if (NetworkPawn != nullptr)
48 {
49 const int PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex;
50 TArray<ANetworkPawn*> NetworkPawns;
51 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
52 {
53 NetworkPawns.Add(*It);
54 }
55 if (NetworkPawns.Num() > 1)
56 {
57 if (PlayerIndex == 0)
58 {
59 SendGgpo(NetworkPawns[1], true);
60 }
61 else
62 {
63 SendGgpo(NetworkPawns[0], false);
64 }

329void ANightSkyPlayerController::SendGgpo(ANetworkPawn* InNetworkPawn, bool Client)
330{
331 if(InNetworkPawn->FighterMultiplayerRunner==nullptr)//TODO: CHECK IF MULTIPLAYERRUNNER IS SPAWNED BEFORE THIS, IF SO DO THIS IN BEGINPLAY
332 {
333 TArray<AActor*> FoundFighterGameStates;
334 UGameplayStatics::GetAllActorsOfClass(GetWorld(), AFighterMultiplayerRunner::StaticClass(), FoundFighterGameStates);
335 if(FoundFighterGameStates.Num()>0)
336 {
337 InNetworkPawn->FighterMultiplayerRunner = Cast<AFighterMultiplayerRunner>(FoundFighterGameStates[0]);
338 }
339 }
340
341 if (InNetworkPawn->FighterMultiplayerRunner && InNetworkPawn->FighterMultiplayerRunner->connectionManager)
342 {
343 while(InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.Num()>0)
344 {
345 const auto SendVal = InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.GetTail();
346 if(Client)
347 {
348 InNetworkPawn->SendGgpoToClient(SendVal->GetValue());
349 }
350 else
351 {
352 InNetworkPawn->SendGgpoToServer(SendVal->GetValue());
353 }
354 InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.Empty();
355 //InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.RemoveNode(SendVal);
356 }
357 }
358}

360void ANightSkyPlayerController::SendBattleData()
361{
362 int PlayerIndex = 0;
363 if (GetWorld()->GetNetMode() == NM_Client)
364 PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex = 1;
365 TArray<ANetworkPawn*> NetworkPawns;
366 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
367 {
368 NetworkPawns.Add(*It);
369 }
370 if (NetworkPawns.Num() > 1)
371 {
372 UNightSkyGameInstance* GameInstance = Cast<UNightSkyGameInstance>(GetGameInstance());
373 if (PlayerIndex == 0)
374 {
375 NetworkPawns[1]->ClientGetBattleData(GameInstance->BattleData);
376 }
377 else
378 {
379 NetworkPawns[0]->ServerGetBattleData(GameInstance->BattleData);
380 }
381 }
