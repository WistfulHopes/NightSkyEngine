40// Called every frame
41void ANightSkyPlayerController::Tick(float DeltaTime)
42{
43 Super::Tick(DeltaTime);
44 if (NetworkPawn != nullptr)
45 {
46 const int PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex;
47 TArray<ANetworkPawn*> NetworkPawns;
48 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
49 {
50 NetworkPawns.Add(*It);
51 }
52 if (NetworkPawns.Num() > 1)
53 {
54 if (PlayerIndex == 0)
55 {
56 SendGgpo(NetworkPawns[1], true);
57 }
58 else
59 {
60 SendGgpo(NetworkPawns[0], false);
61 }
62 }
63 if (NetworkPawn->bRematchAccepted && bRematch)
64 {
65 NetworkPawn->bRematchAccepted = false;
66 bRematch = false;
67 Cast<ANightSkyGameState>(GetWorld()->GetGameState())->MatchInit();
68 }
69 }
70}

277void ANightSkyPlayerController::SendGgpo(ANetworkPawn* InNetworkPawn, bool Client) const
278{
279 if(InNetworkPawn->FighterMultiplayerRunner==nullptr)//TODO: CHECK IF MULTIPLAYERRUNNER IS SPAWNED BEFORE THIS, IF SO DO THIS IN BEGINPLAY
280 {
281 TArray<AActor*> FoundFighterGameStates;
282 UGameplayStatics::GetAllActorsOfClass(GetWorld(), AFighterMultiplayerRunner::StaticClass(), FoundFighterGameStates);
283 if(FoundFighterGameStates.Num()>0)
284 {
285 InNetworkPawn->FighterMultiplayerRunner = Cast<AFighterMultiplayerRunner>(FoundFighterGameStates[0]);

288
289 if (InNetworkPawn->FighterMultiplayerRunner && InNetworkPawn->FighterMultiplayerRunner->connectionManager)

292 {
293 const auto SendVal = InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.GetTail();
294 if(Client)
295 {
296 InNetworkPawn->SendGgpoToClient(SendVal->GetValue());
297 }
298 else
299 {
300 InNetworkPawn->SendGgpoToServer(SendVal->GetValue());
301 }
302 InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.Empty();
303 //InNetworkPawn->FighterMultiplayerRunner->connectionManager->sendSchedule.RemoveNode(SendVal);
304 }
305 }

308void ANightSkyPlayerController::SendBattleData()
309{
310 int PlayerIndex = 0;
311 if (GetWorld()->GetNetMode() == NM_Client)
312 PlayerIndex = Cast<UNightSkyGameInstance>(GetGameInstance())->PlayerIndex = 1;
313 TArray<ANetworkPawn*> NetworkPawns;
314 for (TActorIterator<ANetworkPawn> It(GetWorld()); It; ++It)
315 {
316 NetworkPawns.Add(*It);
317 }
318 if (NetworkPawns.Num() > 1)
319 {
320 FNetworkMirror NetworkMirror{};
321 const UNightSkyGameInstance* GameInstance = Cast<UNightSkyGameInstance>(GetGameInstance());
322 if (PlayerIndex == 0)
323 {
324 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[0]->GetPrimaryAssetId());
325 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[1]->GetPrimaryAssetId());
326 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[2]->GetPrimaryAssetId());
327 NetworkPawns[1]->ClientGetBattleData(GameInstance->BattleData, NetworkMirror);
328 }
329 else
330 {
331 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[3]->GetPrimaryAssetId());
332 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[4]->GetPrimaryAssetId());
333 NetworkMirror.PlayerList.Add(GameInstance->BattleData.PlayerListP1[5]->GetPrimaryAssetId());
334 NetworkPawns[0]->ServerGetBattleData(GameInstance->BattleData, NetworkMirror);
335 }
336 }
