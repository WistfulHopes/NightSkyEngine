25void ANightSkyAIController::Update()
26{
27 if (!Player) Player = GetPawn<APlayerObject>();
28
29 if (GameState->BattleState.TimeUntilRoundStart || Player->PlayerFlags & PLF_RoundWinInputLock || !Player->IsOnScreen())
30 return;
31
32 if (Player->CurrentHealth <= 0) return;
33
34 bBlock = false;
35
36 bool bDefendOnly = false;
37
38 if (Player->Enemy->CheckIsAttacking()
39 && Player->CheckEnemyInRange(
40 Player->Enemy->StoredStateMachine.CurrentState->CPUData.AttackXBeginRange,
41 Player->Enemy->StoredStateMachine.CurrentState->CPUData.AttackXEndRange,
42 Player->Enemy->StoredStateMachine.CurrentState->CPUData.AttackYBeginRange,
43 Player->Enemy->StoredStateMachine.CurrentState->CPUData.AttackYEndRange)
44 && WaitCount <= WaitLimit)
45 {
46 bDefendOnly = true;
47 }
48 else
49 {
50 WaitCount++;
51 if (WaitCount <= WaitLimit) return;
52 }
53
54 TArray<UState*> AvailableStates{};
55 for (auto State : Player->StoredStateMachine.States)
56 {
57 if (!Player->CanEnterState(State)) continue;
58 AvailableStates.Add(State);
59 }
60
61 if (AvailableStates.Num() == 1)
62 {
63 ResetParams();
64 TargetState = AvailableStates[0];
65 }
66 else
67 {
68 auto MaxWeight = 50;
69 UState* CurState = nullptr;
70 for (const auto State : AvailableStates)
71 {
72 int Weight;
73 if (bDefendOnly)
74 {
75 if (!State->CPUData.bAttack)
76 {
77 Weight = CheckDefenseWeight(State);
78 if (Weight > MaxWeight)
79 {
80 MaxWeight = Weight;
81 CurState = State;
82 }
83 }
84 else
85 {
86 Weight = CheckAttackWeight(State);
87 if (Weight > MaxWeight)
88 {
89 MaxWeight = Weight;
90 CurState = State;
91 }
92 }
93 }
94 else if (!State->CPUData.bAttack)
95 {
96 Weight = CheckBasicWeight(State);
97 if (Weight > MaxWeight)
98 {
99 MaxWeight = Weight;
100 CurState = State;
101 }
102 }
103 else
104 {
105 Weight = CheckAttackWeight(State);
106 if (Weight > MaxWeight)
107 {
108 MaxWeight = Weight;
109 CurState = State;
110 }
111 }
112 }
113 if (TargetState != CurState)
114 {
115 ResetParams();
116 TargetState = CurState;
117 }
118 }
119
120 InputCount = FMath::Max(InputCount - InputCountLimit, 0);
121 if (SetInputs()) ResetParams();
122}

132bool ANightSkyAIController::SetInputs()
133{
134 if (!TargetState || TargetState->InputConditionLists.Num() == 0) return false;
135 for (auto Condition : TargetState->InputConditionLists[0].InputConditions)
136 {
137 if (InputCount > InputCountLimit) return false;
138
139 Player->StoredInputBuffer.WriteInputCondition(Condition);
140 InputCount += Condition.Sequence.Num();
141 }
142
143 if (bBlock)
144 {
145 Player->StoredInputBuffer.InputBufferInternal[89] |= INP_Left;
146 }
147
148 TargetState = nullptr;
149 return true;
150}

164int32 ANightSkyAIController::CheckBasicWeight(const UState* State) const
165{
166 if (!IsValid(Player)) return 0;
167
168 int Weight = 0;
169
170 switch (State->StateType)
171 {
172 case EStateType::Standing:
173 Weight = GameState->BattleState.RandomManager.RandRange(25, 50);
174 break;
175 case EStateType::Crouching:
176 Weight = GameState->BattleState.RandomManager.RandRange(25, 50);
177 break;
178 case EStateType::NeutralJump:
179 case EStateType::ForwardJump:
180 Weight = GameState->BattleState.RandomManager.RandRange(5, 10);
181 if (GetEnemyDistanceY() > 105000 && Player->PosY < Player->Enemy->PosY)
182 {
183 Weight += GameState->BattleState.RandomManager.RandRange(5, 15);
184 }
185 break;
186 case EStateType::ForwardAirDash:
187 Weight = GameState->BattleState.RandomManager.RandRange(35, 50);
188 if (GetEnemyDistanceX() > 360000)
189 {
190 Weight += GameState->BattleState.RandomManager.RandRange(15, 30);
191 }
192 break;
193 case EStateType::BackwardJump:
194 case EStateType::BackwardAirDash:
195 Weight = GameState->BattleState.RandomManager.RandRange(10, 15);
196 if (Player->Enemy->CheckIsAttacking() && GetEnemyDistanceX() < 360000)
197 {
198 Weight += GameState->BattleState.RandomManager.RandRange(25, 50);
199 }
200 break;
201 case EStateType::ForwardWalk:
202 Weight = GameState->BattleState.RandomManager.RandRange(20, 80);
203 if (GetEnemyDistanceX() > 360000)
204 {
205 Weight += GameState->BattleState.RandomManager.RandRange(15, 30);
206 }
207 if (Player->Enemy->CheckIsAttacking() && GetEnemyDistanceX() < 240000)
208 {
209 Weight -= GameState->BattleState.RandomManager.RandRange(10, 20);
210 }
211 if (Player->Enemy->CheckIsStunned())
212 {
213 Weight += GameState->BattleState.RandomManager.RandRange(5, 10);
214 }
215 break;
216 case EStateType::ForwardDash:
217 Weight = GameState->BattleState.RandomManager.RandRange(50, 150);
218 if (GetEnemyDistanceX() > 480000)
219 {
220 Weight += GameState->BattleState.RandomManager.RandRange(15, 30);
221 }
222 if (Player->Enemy->CheckIsAttacking() && GetEnemyDistanceX() < 360000)
223 {
224 Weight -= GameState->BattleState.RandomManager.RandRange(10, 20);
225 }
226 if (Player->Enemy->CheckIsStunned())
227 {
228 Weight += GameState->BattleState.RandomManager.RandRange(5, 10);
229 }
230 break;
231 case EStateType::BackwardWalk:
232 Weight = GameState->BattleState.RandomManager.RandRange(10, 65);
233 if (Player->Enemy->CheckIsAttacking() && GetEnemyDistanceX() < 360000)
234 {
235 Weight += GameState->BattleState.RandomManager.RandRange(15, 35);
236 }
237 if (Player->Enemy->CheckIsStunned())
238 {
239 Weight -= GameState->BattleState.RandomManager.RandRange(10, 20);
240 }
241 break;
242 case EStateType::BackwardDash:
243 Weight = GameState->BattleState.RandomManager.RandRange(20, 40);
244 if (Player->Enemy->CheckIsAttacking() && GetEnemyDistanceX() < 360000)
245 {
246 Weight += GameState->BattleState.RandomManager.RandRange(10, 20);
247 }
248 if (Player->Enemy->CheckIsStunned())
249 {
250 Weight -= GameState->BattleState.RandomManager.RandRange(25, 50);
251 }
252 break;
253 case EStateType::Tech:
254 // TODO tech weight
255 Weight = 250;
256 break;
257 default:
258 break;
259 }
260
261 Weight += GameState->BattleState.RandomManager.RandRange(-50, 50);
262
263 return Weight;
264}

266int32 ANightSkyAIController::CheckAttackWeight(const UState* State) const
267{
268 if (!IsValid(Player)) return 0;
269
270 if (!Player->CheckEnemyInRange(State->CPUData.AttackXBeginRange, State->CPUData.AttackXEndRange,
271 State->CPUData.AttackYBeginRange, State->CPUData.AttackYEndRange)
272 && !State->CPUData.bProjectile)
273 return 0;
274
275 int32 Weight = GameState->BattleState.RandomManager.RandRange(80, 150);
276
277 if (Player->PosY < Player->Enemy->PosY && State->CPUData.bAntiAir) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
278
279 if (State->CPUData.bUsesResource) Weight -= GameState->BattleState.RandomManager.RandRange(15, 40);
280 if (State->CPUData.bBigDamage) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
281 if (State->CPUData.bNoCombo && Player->ComboCounter > 2) Weight -= GameState->BattleState.RandomManager.RandRange(15, 40);
282
283 switch (State->CPUData.AttackSpeed)
284 {
285 case ASPD_Fast:
286 Weight += GameState->BattleState.RandomManager.RandRange(25, 50);
287 break;
288 case ASPD_Medium:
289 break;
290 case ASPD_Slow:
291 Weight -= GameState->BattleState.RandomManager.RandRange(15, 30);
292 break;
293 default: break;
294 }
295
296 if (Player->Enemy->StoredStateMachine.CurrentState->StateType == EStateType::Hitstun)
297 {
298 if (State->CPUData.bThrow && !State->CPUData.bCombo) Weight = 0;
299 if (State->CPUData.bCombo) Weight += GameState->BattleState.RandomManager.RandRange(35, 60);
300 }
301 else if (Player->IsEnemyBlocking())
302 {
303 if (State->CPUData.bBlockstring) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
304 if (Player->Enemy->GetCurrentStateName() == State_Universal_StandBlock)
305 {
306 if (State->CPUData.BlockType == BLK_Low) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
307 }
308 else if (Player->Enemy->GetCurrentStateName() == State_Universal_CrouchBlock)
309 {
310 if (State->CPUData.BlockType == BLK_High) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
311 }
312 else if (State->CPUData.BlockType == BLK_None) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
313
314 if (State->CPUData.bThrow)
315 {
316 if (Player->Enemy->CheckIsStunned()) Weight = 0;
317 else Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
318 }
319 }
320 else
321 {
322 if (Player->Enemy->IsEnemyAttackState() && State->CPUData.bPunish)
323 {
324 bool bInPunishRange = false;
325 switch (State->CPUData.PunishRange)
326 {
327 case RAN_Near:
328 bInPunishRange = GetEnemyDistanceX() <= 420000;
329 break;
330 case RAN_Mid:
331 bInPunishRange = GetEnemyDistanceX() > 420000 && GetEnemyDistanceX() <= 840000;
332 break;
333 case RAN_Far:
334 bInPunishRange = GetEnemyDistanceX() > 840000;
335 break;
336 default:
337 break;
338 }
339
340 Weight += bInPunishRange ? GameState->BattleState.RandomManager.RandRange(15, 40) : 0;
341 }
342
343 if (State->CPUData.bInvuln) Weight += GameState->BattleState.RandomManager.RandRange(15, 40);
344 }
345 Weight += GameState->BattleState.RandomManager.RandRange(-50, 50);
346
347 return Weight;
348}
