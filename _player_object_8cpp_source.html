115uint32 APlayerObject::FlipInput(uint32 Input)
116{
117 const unsigned int Bit1 = Input >> 2 & 1;
118 const unsigned int Bit2 = Input >> 3 & 1;
119 unsigned int x = Bit1 ^ Bit2;
120
121 x = x << 2 | x << 3;
122
123 return Input ^ x;
124}

150bool APlayerObject::HandleAutoCombo(int32 StateIndex)
151{
152 if (!FindAutoComboCancelOption(StoredStateMachine.States[StateIndex]->Name)) return false;
153
154 bool AutoComboSuccess = false;
155 for (int i = 0; i < 8; i++)
156 {
157 const auto AutoComboCancel = AutoComboCancels[i];
158 if (AutoComboCancel != StateIndex) continue;
159
160 EInputFlags Button;
161 switch (i)
162 {
163 case 0:
164 Button = INP_A;
165 break;
166 case 1:
167 Button = INP_A;
168 break;
169 case 2:
170 Button = INP_A;
171 break;
172 case 3:
173 Button = INP_A;
174 break;
175 case 4:
176 Button = INP_A;
177 break;
178 case 5:
179 Button = INP_A;
180 break;
181 case 6:
182 Button = INP_A;
183 break;
184 case 7:
185 Button = INP_A;
186 break;
187 default:
188 Button = INP_A;
189 break;
190 }
191
192 FInputCondition AutoComboCondition;
193 AutoComboCondition.Sequence.Add(FInputBitmask(Button));
194 AutoComboCondition.Sequence.Last().Lenience = 0;
195 AutoComboCondition.Method = EInputMethod::Once;
196
197 if (CheckInput(AutoComboCondition))
198 {
199 AutoComboSuccess = true;
200 break;
201 }
202 }
203
204 if (!AutoComboSuccess) return false;
205
206 bIsAutoCombo = HandleStateTransition(StateIndex, true);
207 return bIsAutoCombo;

210bool APlayerObject::HandleStateInputs(int32 StateIndex, bool Buffer)
211{
212 for (FInputConditionList& List : StoredStateMachine.States[StateIndex]->InputConditionLists)
213 {
214 for (int v = 0; v < List.InputConditions.Num(); v++) //iterate over input conditions
215 {
216 //check input condition against input buffer, if not met break.
217 if (!StoredInputBuffer.CheckInputCondition(List.InputConditions[v],
218 (CancelFlags & CNC_EnableKaraCancel) == 0))
219 {
220 break;
221 }
222 if (v == List.InputConditions.Num() - 1) //have all conditions been met?
223 {
224 return HandleStateTransition(StateIndex, Buffer);
225 }
226 }
227 if (List.InputConditions.Num() == 0) //if no input conditions, set state
228 {
229 return HandleStateTransition(StateIndex, Buffer);
230 }
231 }
232 return false;
233}

235bool APlayerObject::HandleStateTransition(int32 StateIndex, bool Buffer)
236{
237 if (FindChainCancelOption(StoredStateMachine.States[StateIndex]->Name)
238 || FindAutoComboCancelOption(StoredStateMachine.States[StateIndex]->Name)
239 || FindWhiffCancelOption(StoredStateMachine.States[StateIndex]->Name)
240 || CancelFlags & CNC_CancelIntoSelf) //if cancel option, allow resetting state
241 {
242 if (Buffer)
243 {
244 BufferedStateName = StoredStateMachine.States[StateIndex]->Name;
245 return true; //don't try to enter another state
246 }
247 if (StoredStateMachine.ForceSetState(StoredStateMachine.States[StateIndex]->Name)) //if state set successful...
248 {
249 GotoLabelActive = false;
250 switch (StoredStateMachine.States[StateIndex]->EntryStance)
251 {
252 case EEntryStance::Standing:
253 Stance = ACT_Standing;
254 break;
255 case EEntryStance::Crouching:
256 Stance = ACT_Crouching;
257 break;
258 case EEntryStance::Jumping:
259 Stance = ACT_Jumping;
260 break;
261 default:
262 break;
263 }
264 return true; //don't try to enter another state
265 }
266 }
267 else
268 {
269 if (Buffer)
270 {
271 BufferedStateName = StoredStateMachine.States[StateIndex]->Name;
272 return true; //don't try to enter another state
273 }
274 if (StoredStateMachine.SetState(StoredStateMachine.States[StateIndex]->Name)) //if state set successful...
275 {
276 GotoLabelActive = false;
277 switch (StoredStateMachine.States[StateIndex]->EntryStance)
278 {
279 case EEntryStance::Standing:
280 Stance = ACT_Standing;
281 break;
282 case EEntryStance::Crouching:
283 Stance = ACT_Crouching;
284 break;
285 case EEntryStance::Jumping:
286 Stance = ACT_Jumping;
287 break;
288 default:
289 break;
290 }
291 return true; //don't try to enter another state
292 }
293 }
294 return false; //state couldn't be entered
295}

716void APlayerObject::HandleHitAction(EHitAction HACT)
717{
718 int32 FinalHitstop = ReceivedHit.Hitstop + ReceivedHit.EnemyHitstopModifier;
719
720 AttackOwner->Hitstop = ReceivedHit.Hitstop;
721 Hitstop = FinalHitstop;
722
723 int32 Proration = ReceivedHit.ForcedProration;
724 if (Enemy->ComboCounter == 0)
725 Proration *= ReceivedHit.InitialProration;
726 else
727 Proration *= 100;
728 if (Enemy->ComboCounter == 0)
729 TotalProration = 10000;
730 Proration = Proration * TotalProration / 10000;
731
732 if ((AttackFlags & ATK_ProrateOnce) == 0 || (AttackFlags & ATK_ProrateOnce && Enemy->ComboCounter == 0))
733 TotalProration = Proration;
734 else
735 Proration = TotalProration;
736
737 Enemy->ComboCounter++;
738
739 if (ReceivedHitCommon.DeathCamOverride)
740 PlayerFlags |= PLF_DeathCamOverride;
741
742 else
743 PlayerFlags &= ~PLF_DeathCamOverride;
744
745 if (PlayerFlags & PLF_IsKnockedDown)
746 {
747 if (PosY <= GroundHeight)
748 PosY = GroundHeight + 1;
749
750 if (!(AttackOwner->AttackFlags & ATK_IgnoreOTG))
751 OTGCount++;
752 PlayerFlags &= ~PLF_IsKnockedDown;
753 }
754
755 int FinalDamage;
756 if (Enemy->ComboCounter == 1)
