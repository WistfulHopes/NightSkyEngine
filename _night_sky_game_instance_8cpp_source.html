280void UNightSkyGameInstance::HandleFindSessionsComplete(
281 bool bArg, TSharedRef<FOnlineSessionSearch, ESPMode::ThreadSafe> Shared)
282{
283 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
284 IOnlineSessionPtr Session = Subsystem->GetSessionInterface();
285
286 if (bArg)
287 {
288 for (auto RawResult : Shared->SearchResults)
289 {
290 if (RawResult.IsValid())
291 {
292 FString ConnectInfo;
293 if (Session->GetResolvedConnectString(RawResult, NAME_GamePort, ConnectInfo))
294 {
295 FSessionInfo SessionInfo;
296 SessionInfo.Result = RawResult;
297 SessionInfo.ConnectInfo = ConnectInfo;
298
299 bool Duplicate = false;
300
301 for (auto StoredSession : SessionInfos)
302 {
303 if (StoredSession.ConnectInfo == ConnectInfo)
304 {
305 Duplicate = true;
306 break;
307 }
308 }
309
310 if (!Duplicate)
311 SessionInfos.Add(SessionInfo);
312 }
313 }
314 }
315 }
316
317 UE_LOG(LogTemp, Warning, TEXT("Session count: %d"), SessionInfos.Num())
318
319 Session->ClearOnFindSessionsCompleteDelegate_Handle(this->FindSessionsDelegateHandle);
320 this->FindSessionsDelegateHandle.Reset();
