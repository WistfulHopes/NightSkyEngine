119void UNightSkyGameInstance::FindReplays()
120{
121 ReplayList.Empty();
122 for (int i = 0; i < MaxReplays; i++)
123 {
124 FString ReplayName = "REPLAY";
125 ReplayName.AppendInt(i);
126 if (!UGameplayStatics::DoesSaveGameExist(ReplayName, 0))
127 {
128 continue;
129 }
130 ReplayList.Add(Cast<UReplaySaveInfo>(UGameplayStatics::LoadGameFromSlot(ReplayName, 0)));
131 ReplayList.Last()->ReplayIndex = i;
132 if (ReplayList.Last()->Version != BattleVersion)
133 {
134 ReplayList.Pop();
135 UGameplayStatics::DeleteGameInSlot(ReplayName, 0);
136 }
137 }
138 BP_OnFindReplaysComplete(ReplayList);
